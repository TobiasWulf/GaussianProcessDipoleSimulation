	\begin{itemize}
		\item Anwendungsfall Drehzahlmessung
		\item Anwendungsfall Winkelmessung
		\item Ausblick TMR und Strommessung
		\item Aufbau Sensorbrücke TMR (Umriss)
	\end{itemize}
